# Получение информации о карте по токену

openapi: 3.0.1
info:
  title: ''
  description: ''
  version: 1.0.0
paths:
  /payment/card/{card_token}:
    get:
      summary: Получение информации о карте по токену
      deprecated: false
      description: ''
      tags:
        - Привязка карт (форма)
      parameters:
        - name: card_token
          in: path
          description: Идентификатор добавленной карты в систему

          example: 6225f3c93f7a880142782fa4
          schema:
            type: string
      requestBody:
        content:
          application/json:
            schema:
              type: object
              properties: {}
            examples: {}
      responses:
        '200':
          description: ''
          content:
            application/json:
              schema:
                type: object
                properties:
                  success:
                    type: boolean
                  data:
                    type: object
                    properties:
                      id:
                        type: integer
                      application_id:
                        type: integer
                        description: >-
                          Идентификатор приложения, выданный со стороны
                          Multicard
                      payer_id:
                        type: string
                        description: >+
                          Значение поля session_id из ответа на запрос привязки
                          карты

                      card_pan:
                        type: string
                        description: |-
                          Max length:16
                          Маскированный номер карты
                      card_token:
                        type: string
                        description: |-
                          Max length:255
                          Токен карты для проведения платежных операций с картой
                      phone:
                        type: string
                        description: >-
                          Max length:12

                          Телефон держателя карты в формате 998XXXXXXXXX
                          (поддерживается только для карт Uzcard и Humo)
                      holder_name:
                        type: string
                        description: |-
                          Max length:100
                          Имя держателя карты
                      pinfl:
                        type: 'null'
                        description: >-
                          Default:ПИНФЛ держателя карты (возвращается в случае
                          передачи в запросе на создание привязки для карт
                          Uzcard и Humo)

                          Max length:14
                      ps:
                        type: string
                        description: |+
                          Allowed values (5)
                          uzcard,
                          humo,
                          visa,
                          mastercard,
                          unionpay
                          hide all
                          Платежная система

                      status:
                        type: string
                        description: >-
                          Allowed values (3)

                          active,

                          draft,

                          deleted

                          hide all

                          Статус привязки. active - карта успешно привязана,
                          можно проводить платежные операции по токену карты
                      added_on:
                        type: string
                        description: |+
                          Время добавления карты в формате YYYY-mm-dd H:i:s

                      card_status:
                        type: object
                        properties:
                          code:
                            type: integer
                            description: |+
                              Состояние карты. 0 - карта активна

                          description:
                            type: string
                            description: |+
                              Описание блокировки карты

                        required:
                          - code
                          - description
                        x-apidog-orders:
                          - code
                          - description
                        description: >-
                          Текущий статус карты (например, активна или
                          заблокирована). Содержит код и текстовое описание.
                      sms_inform:
                        type: boolean
                        description: >-
                          Признак подключения SMS-уведомлений (true = включены,

                    required:
                      - id
                      - application_id
                      - payer_id
                      - card_pan
                      - card_token
                      - phone
                      - holder_name
                      - pinfl
                      - ps
                      - status
                      - added_on
                      - card_status
                      - sms_inform
                    x-apidog-orders:
                      - id
                      - application_id
                      - payer_id
                      - card_pan
                      - card_token
                      - phone
                      - holder_name
                      - pinfl
                      - ps
                      - status
                      - added_on
                      - card_status
                      - sms_inform
                required:
                  - success
                  - data
                x-apidog-orders:
                  - success
                  - data
              example:
                success: true
                data:
                  id: 55
                  application_id: 1
                  payer_id: '2313122'
                  card_pan: 409784******3066
                  card_token: 6225f3c93f7a110142782fa4
                  phone: '998901234567'
                  holder_name: AAAA BBBBB
                  pinfl: null
                  ps: uzcard
                  status: active
                  added_on: '2022-04-21 14:36:29'
                  card_status:
                    code: 0
                    description: Active
                  sms_inform: true
          headers: {}
          x-apidog-name: Проверка привязанной карты
        '400':
          description: >-
            При поиске несуществуюещей карты, возвращается ошибка с кодом 400,
            также с текстом
          content:
            application/json:
              schema:
                type: object
                properties:
                  success:
                    type: boolean
                    description: Статус запроса
                  error:
                    type: object
                    properties:
                      code:
                        type: string
                        description: Кодовое сообщение
                      details:
                        type: string
                        description: Описание ошибки
                    required:
                      - code
                      - details
                    x-apidog-orders:
                      - code
                      - details
                    description: Ошибка
                required:
                  - success
                  - error
                x-apidog-orders:
                  - success
                  - error
              example:
                success: false
                error:
                  code: ERROR_CARD_NOT_FOUND
                  details: Неверный номер карты или срок действия
          headers: {}
          x-apidog-name: 'Поиск несуществующей карты '
      security: []
      x-apidog-folder: Привязка карт (форма)
      x-apidog-status: released
      x-run-in-apidog: https://app.apidog.com/web/project/1022226/apis/api-19729305-run
components:
  schemas: {}

  - bearer: []
